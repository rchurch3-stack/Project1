recode_factors <- function(data) {

  if ("SEX" %in% names(data)) {
    data$SEX <- factor(
      data$SEX,
      levels = c("1", "2"),
      labels = c("Male", "Female")
    )
  }

  data
}
parse_pums <- function(response) {

  # Extract the API response as text
  response_text <- content(response, "text")

  # Convert the JSON response into an R object
  parsed <- fromJSON(response_text)

  # Remove the first row and convert the remaining data to a tibble
  pums_data <- as_tibble(parsed[-1, ])

  # Use the first row as the column names
  colnames(pums_data) <- parsed[1, ]

  # Return the cleaned tibble
  pums_data
}

query_pums <- function(
    year = 2022,
    numeric_vars = c("AGEP", "PWGTP"),
    categorical_vars = c("SEX"),
    geography = "state",
    subset_value = "37", # North Carolina by default
    key
) {

  # --------------------------
  # Validate year
  # --------------------------

  if (!is.numeric(year) || length(year) != 1) {
    stop("year must be a single numeric value.")
  }

  if (!(year %in% 2010:2022)) {
    stop("year must be between 2010 and 2022.")
  }

  # --------------------------
  # Validate numeric variables
  # --------------------------

  valid_numeric <- c(
    "AGEP",
    "GASP",
    "GRPIP",
    "JWAP",
    "JWDP",
    "JWMNP",
    "PWGTP"
  )

  if (!all(numeric_vars %in% valid_numeric)) {
    stop("One or more numeric variables are invalid.")
  }

  # Ensure PWGTP is always returned
  if (!("PWGTP" %in% numeric_vars)) {
    numeric_vars <- c(numeric_vars, "PWGTP")
  }

  # Require at least one numeric variable besides PWGTP
  if (length(setdiff(numeric_vars, "PWGTP")) < 1) {
    stop("At least one numeric variable besides PWGTP must be selected.")
  }

  # --------------------------
  # Validate categorical variables
  # --------------------------

  valid_categorical <- c(
    "FER",
    "HHL",
    "SCH",
    "SCHL",
    "SEX"
  )

  if (!all(categorical_vars %in% valid_categorical)) {
    stop("One or more categorical variables are invalid.")
  }

  if (length(categorical_vars) < 1) {
    stop("At least one categorical variable must be selected.")
  }

  # --------------------------
  # Validate geography
  # --------------------------

  geography <- tolower(geography)

  valid_geography <- c(
    "region",
    "division",
    "state"
  )

  if (!(geography %in% valid_geography)) {
    stop("geography must be region, division, or state.")
  }

  # --------------------------
  # Build variable list
  # --------------------------

  variables <- unique(
    c(
      numeric_vars,
      categorical_vars
    )
  )

  variable_string <- paste(
    variables,
    collapse = ","
  )

  # --------------------------
  # Geography filter
  # --------------------------

  if (tolower(subset_value) == "all") {

    geography_clause <- paste0(
      "&for=",
      geography,
      ":*"
    )

  } else {

    geography_clause <- paste0(
      "&for=",
      geography,
      ":",
      subset_value
    )
  }

  # --------------------------
  # Build URL
  # --------------------------

  url <- paste0(
    "https://api.census.gov/data/",
    year,
    "/acs/acs1/pums?",
    "get=",
    variable_string,
    geography_clause,
    "&key=",
    key
  )

  # --------------------------
  # Query API
  # --------------------------

  response <- GET(url)

  # --------------------------
  # Convert to tibble
  # --------------------------
  
  # Convert response to tibble
  data <- parse_pums(response)
  
  numeric_columns <- intersect(
  names(data),
  c(
    "AGEP",
    "GASP",
    "GRPIP",
    "JWAP",
    "JWDP",
    "JWMNP",
    "PWGTP"
  )
)

data <- data |>
  mutate(
    across(
      all_of(numeric_columns),
      as.numeric
    )
  )

factor_columns <- intersect(
  names(data),
  c(
    "FER",
    "HHL",
    "SCH",
    "SCHL",
    "SEX"
  )
)

data <- data |>
  mutate(
    across(
      all_of(factor_columns),
      as.factor
    )
  )

  # make the SEX variable more readable by assigning Male and Female instead of 1 and 2
  data <- recode_factors(data)

  # Add custom census class
  class(data) <- c("census", class(data))

  return(data)
}

Next, we write a method capable of calling our single year function as many times as needed:

query_pums_years <- function(
    years,
    numeric_vars = c("AGEP", "PWGTP"),
    categorical_vars = c("SEX"),
    geography = "state",
    subset_value = "37",
    key
) {

  # Remove 2020
  if (2020 %in% years) {
    stop("2020 is not allowed because Census discarded that data.")
  }

  results <- lapply(
    years,
    function(y) {

      query_pums(
        year = y,
        numeric_vars = numeric_vars,
        categorical_vars = categorical_vars,
        geography = geography,
        subset_value = subset_value,
        key = key
      ) |>
        mutate(year = y)

    }
  )

  bind_rows(results)

}

After querying the Census API, we assign the returned tibble an additional class of “census”. This allows us to create custom methods later in the project, such as specialized summary or plotting functions that are automatically used whenever a census object is supplied.

We’d like to now build a custom method for “census” that will return the mean and sd for numerics and return the counts for each of the categorical variables (factors).

summary.census <- function(
    object,
    numeric_vars = NULL,
    categorical_vars = NULL,
    ...
) {

  if (is.null(numeric_vars)) {

    numeric_vars <- names(object)[sapply(object, is.numeric)]

    #Exclude also the time values such as JWAP and JWDP
    numeric_vars <- setdiff(
      numeric_vars,
      c(
        "PWGTP",
        "year",
        "JWAP",
        "JWDP"
      )
    )

  }

  if (is.null(categorical_vars)) {

    categorical_vars <- names(object)[sapply(object, is.factor)]

  }

  results <- list()

  # Weighted summaries
  for (var in numeric_vars) {

    x <- object[[var]]
    w <- object$PWGTP

    weighted_mean <- sum(x * w) / sum(w)

    weighted_sd <- sqrt(
      sum(x^2 * w) / sum(w) -
      weighted_mean^2
    )

    results[[var]] <- list(
      mean = weighted_mean,
      sd = weighted_sd
    )

  }

  # Categorical counts
  for (var in categorical_vars) {

    results[[var]] <- table(
      object[[var]]
    )

  }

  return(results)

}
