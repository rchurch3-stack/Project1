Introduction

The goal of this project is to build reusable functions that interact with the Census Public Use Microdata Sample (PUMS) API. Rather than manually constructing API requests each time, I developed functions that validate user inputs, retrieve data from the API, convert the returned JSON into a tidy tibble, and apply appropriate data types. I also created custom summary and plotting methods for census data objects so that common analyses can be performed consistently across multiple datasets.

Helper Functions

The Census API returns data in JSON format. To make the data easier to work with in R, I first created helper functions that convert API responses into tibbles and recode categorical variables into more meaningful labels.

 # --------------------------
  # Helper methods
  # --------------------------

recode_factors <- function(data) {

  if ("SEX" %in% names(data)) {
    data$SEX <- factor(
      data$SEX,
      levels = c("1", "2"),
      labels = c("Male", "Female")
    )
  }
  
  if ("SCH" %in% names(data)) {
    data$SCH <- factor(
      data$SCH,
      levels = c("0", "1", "2", "3"),
      labels = c(
        "Under 3",
        "Not Enrolled",
        "Public School",
        "Private/Home School"
      )
    )
  }

  data
}

# Census encodes JWAP/JWDP as time intervals.
# These are converted to time values and are not
# included in summary.census().
convert_time_variable <- function(x) {

  x <- as.numeric(x)

  x[x == 0] <- NA

  x
}

The parse_pums() function extracts the JSON response, converts it into an R object, and transforms the returned data into a tidy tibble with appropriate column names.

parse_pums <- function(response) {

  # Extract the API response as text
  response_text <- content(response, "text")

  # Convert the JSON response into an R object
  parsed <- fromJSON(response_text)

  # Remove the first row and convert the remaining data to a tibble
  pums_data <- as_tibble(parsed[-1, ])

  # Use the first row as the column names
  colnames(pums_data) <- parsed[1, ]

  # Return the cleaned tibble
  pums_data
}

Querying the Census API

The primary function developed for this project is query_pums(). This function allows a user to specify a survey year, numeric variables, categorical variables, geography level, and geographic subset. Input validation is performed to ensure that only supported years and variables are requested. The function also guarantees that the survey weight variable (PWGTP) is always included.

query_pums <- function(
    year = 2022,
    numeric_vars = c("AGEP", "PWGTP"),
    categorical_vars = c("SEX"),
    geography = "state",
    subset_value = "37", # North Carolina by default
    key
) {

  # --------------------------
  # Validate year
  # --------------------------

  if (!is.numeric(year) || length(year) != 1) {
    stop("year must be a single numeric value.")
  }

  if (!(year %in% 2010:2022)) {
    stop("year must be between 2010 and 2022.")
  }

  # --------------------------
  # Validate numeric variables
  # --------------------------

  valid_numeric <- c(
    "AGEP",
    "GASP",
    "GRPIP",
    "JWAP",
    "JWDP",
    "JWMNP",
    "PWGTP"
  )

  if (!all(numeric_vars %in% valid_numeric)) {
    stop("One or more numeric variables are invalid.")
  }

  # Ensure PWGTP is always returned
  if (!("PWGTP" %in% numeric_vars)) {
    numeric_vars <- c(numeric_vars, "PWGTP")
  }

  # Require at least one numeric variable besides PWGTP
  if (length(setdiff(numeric_vars, "PWGTP")) < 1) {
    stop("At least one numeric variable besides PWGTP must be selected.")
  }

  # --------------------------
  # Validate categorical variables
  # --------------------------

  valid_categorical <- c(
    "FER",
    "HHL",
    "SCH",
    "SCHL",
    "SEX"
  )

  if (!all(categorical_vars %in% valid_categorical)) {
    stop("One or more categorical variables are invalid.")
  }

  if (length(categorical_vars) < 1) {
    stop("At least one categorical variable must be selected.")
  }

  # --------------------------
  # Validate geography
  # --------------------------

  geography <- tolower(geography)

  valid_geography <- c(
    "region",
    "division",
    "state"
  )

  if (!(geography %in% valid_geography)) {
    stop("geography must be region, division, or state.")
  }

  # --------------------------
  # Build variable list
  # --------------------------

  variables <- unique(
    c(
      numeric_vars,
      categorical_vars
    )
  )

  variable_string <- paste(
    variables,
    collapse = ","
  )

  # --------------------------
  # Geography filter
  # --------------------------

  if (tolower(subset_value) == "all") {

    geography_clause <- paste0(
      "&for=",
      geography,
      ":*"
    )

  } else {

    geography_clause <- paste0(
      "&for=",
      geography,
      ":",
      subset_value
    )
  }

  # --------------------------
  # Build URL
  # --------------------------

  url <- paste0(
    "https://api.census.gov/data/",
    year,
    "/acs/acs1/pums?",
    "get=",
    variable_string,
    geography_clause,
    "&key=",
    key
  )

  # --------------------------
  # Query API
  # --------------------------

  response <- GET(url)

  # --------------------------
  # Convert to tibble
  # --------------------------
  
  # Convert response to tibble
  data <- parse_pums(response)
  
  numeric_columns <- intersect(
  names(data),
  c(
    "AGEP",
    "GASP",
    "GRPIP",
    "JWAP",
    "JWDP",
    "JWMNP",
    "PWGTP"
  )
)

data <- data |>
  mutate(
    across(
      all_of(numeric_columns),
      as.numeric
    )
  )

factor_columns <- intersect(
  names(data),
  c(
    "FER",
    "HHL",
    "SCH",
    "SCHL",
    "SEX"
  )
)

data <- data |>
  mutate(
    across(
      all_of(factor_columns),
      as.factor
    )
  )

  # make the SEX variable more readable by assigning Male and Female instead of 1 and 2
  data <- recode_factors(data)

  # Add custom census class
  class(data) <- c("census", class(data))

  return(data)
}

Combining Multiple Years

Because many analyses require multiple years of data, I created a wrapper function that repeatedly calls query_pums() and combines the results into a single tibble. The function excludes the year 2020 because the Census Bureau advises against using those data due to disruptions caused by the COVID-19 pandemic.

query_pums_years <- function(
    years,
    numeric_vars = c("AGEP", "PWGTP"),
    categorical_vars = c("SEX"),
    geography = "state",
    subset_value = "37", # North Carolina by default
    key
) {

  # Remove 2020
  if (2020 %in% years) {
    stop("2020 is not allowed because Census discarded that data.")
  }

  results <- lapply(
    years,
    function(y) {

      query_pums(
        year = y,
        numeric_vars = numeric_vars,
        categorical_vars = categorical_vars,
        geography = geography,
        subset_value = subset_value,
        key = key
      ) |>
        mutate(year = y)

    }
  )

  bind_rows(results)

}

Custom Census Class

To support custom methods, the returned tibble is assigned an additional class called “census”. This allows generic functions such as summary() and plot() to automatically use methods specifically designed for census data.

A custom summary method was created for census objects. Numeric variables are summarized using weighted means and weighted standard deviations based on the Census survey weight (PWGTP). Categorical variables are summarized using frequency counts. By default, all eligible variables are summarized, although the user may specify a subset of variables if desired.

summary.census <- function(
    object,
    numeric_vars = NULL,
    categorical_vars = NULL,
    ...
) {

  if (is.null(numeric_vars)) {

    numeric_vars <- names(object)[sapply(object, is.numeric)]

    # JWAP and JWDP are Census-coded time intervals.
    # These variables are excluded from summary statistics
    # because the project instructions allow time variables
    # to be skipped.
    numeric_vars <- setdiff(
      numeric_vars,
      c(
        "PWGTP",
        "year",
        "JWAP",
        "JWDP"
      )
    )

  }

  if (is.null(categorical_vars)) {

    categorical_vars <- names(object)[sapply(object, is.factor)]

  }

  results <- list()

  # Weighted summaries
  for (var in numeric_vars) {

    x <- object[[var]]
    w <- object$PWGTP

    # Handle missing values if any occur.
    weighted_mean <- sum(x * w, na.rm = TRUE) /
      sum(w[!is.na(x)])
    
    weighted_sd <- sqrt(
      sum(x^2 * w, na.rm = TRUE) /
        sum(w[!is.na(x)]) -
        weighted_mean^2
    )

    results[[var]] <- list(
      mean = weighted_mean,
      sd = weighted_sd
    )

  }

  # Categorical counts
  for (var in categorical_vars) {

    results[[var]] <- table(
      object[[var]]
    )

  }

  return(results)

}

Census Plot Method

A custom plotting method was also developed for census objects. The user specifies a categorical variable and a numeric variable, and the function produces a weighted boxplot using the Census survey weights. This provides a simple way to compare distributions across demographic groups.

plot.census <- function(
    x,
    cat_var,
    num_var,
    ...
) {

  ggplot(
    x,
    aes(
      x = get(cat_var),
      y = get(num_var),
      weight = PWGTP
    )
  ) +
    geom_boxplot() +
    labs(
      x = cat_var,
      y = num_var,
      title = paste(
        num_var,
        "by",
        cat_var
      )
    )

}

Exploring the Data

To demonstrate the functions developed in this project, I explored the relationship between age (AGEP) and school enrollment status (SCH). The results show that people currently attending school are generally younger than those who are not enrolled. This example illustrates how the API query, summary, and plotting functions can be combined to quickly investigate patterns in Census data.
